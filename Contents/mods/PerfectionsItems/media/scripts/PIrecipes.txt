module PerfectionsItems
{
    imports
    {
        Base,
    }

    /* Simple recipe for Wooden Sword */
    recipe Make Wooden Sword
    {
        Base.Plank=2,
        Base.Twine=2,
        keep Base.Hammer,
        keep Base.Saw,

        Result:PerfectionsItems.WoodenSword,
        Time:80.0,
        SkillRequired:Woodwork=4,
        Category:Perfection's Items,
        NeedToBeLearn:true,
    }

    /* Stronger recipe for Bokuto - requires higher skill */
    recipe Make Bokuto
    {
        Base.Plank=3,
        Base.Nails=2,
        Base.SmallSheetMetal=1,
        Base.Twine=2,
        keep Base.Hammer,
        keep Base.Saw,

        Result:PerfectionsItems.Bokuto,
        Time:120.0,
        SkillRequired:Woodwork=8,
        Category:Perfection's Items,
        NeedToBeLearn:true,
    }
}
